// GEÇICI: Import işlemi için güvenlik kuralları
// Import tamamlandıktan sonra bunları değiştirin!

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // GEÇICI: Import için tüm yazma işlemlerine izin ver
    // Import tamamlandıktan sonra bunu kaldırın!
    match /{document=**} {
      allow read, write: if true;
    }
  }
}
